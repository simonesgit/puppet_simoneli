Epics which start date or end date is within <data_from> and <data_to>
project = "AAA" AND
((startdate >= "<data_from>" AND startdate <= "<data_to>") OR
(enddate >= "<data_from>" AND enddate <= "<data_to>"))

project = "AAA" AND resolution = Unresolved AND DUE < Now()

project = "AAA" AND (startdate = "" OR enddate = "")
